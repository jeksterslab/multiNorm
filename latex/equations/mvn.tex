\begin{equation}
    \mathcal{N}
    \left(
      \boldsymbol{\mu},
      \boldsymbol{\Sigma}
    \right)
    \label{eq:mvn}
\end{equation}

\noindent where
$\boldsymbol{\mu}$ is a $k \times 1$
vector of means,
and
$\boldsymbol{\Sigma}$ is a $k \times k$
matrix of variances on the diagonals
and covariances on the off-diagonals.
